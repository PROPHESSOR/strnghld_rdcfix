actor HoloDoomImp
{
//$Category Stronghold_HoloMons
  spawnid 255
  health 60
  radius 20
  height 56
  mass 100
  renderstyle Add
  alpha 1
  speed 8
  painchance 200
  PainChance "Stunner", 255
  seesound "imp/sight"
  painsound "imp/pain"
  deathsound "imp/death"
  activesound "imp/active"
  Tag "$STR_MONSTERTAG_HOLODOOMIMP"
  MONSTER
  +FLOORCLIP
  +NOBLOOD
  +NOICEDEATH
  states
  {
  Spawn:
    HROO AB 10 A_Look
    loop
  See:
    HROO AABBCCDD 3 A_Chase
    loop
  Melee:
  Missile:
    HROO EF 8 A_FaceTarget
    HROO G 6 A_CustomComboAttack("HoloDoomImpBall",32,0,"") //A_CustomMissile("HoloDoomImpBall",32,0,0)
    goto See
  Pain:
    HROO H 2
    HROO H 2 A_Pain
    goto See
  Pain.Stunner:
    HROO H 2 A_ChangeFlag("NoPain", 1)
    HROO H 300 A_Pain
    HROO H 0 A_ChangeFlag("NoPain", 0)
    goto See
  Death:
    HROO I 8
    HROO J 8 A_Scream
    HROO K 6
    HROO L 6 A_NoBlocking
    HROO M 40
    HROO M 1 A_FadeOut(0.05)
    wait
  XDeath:
    HROO N 5
    HROO O 5 A_XScream
    HROO P 5
    HROO Q 5 A_NoBlocking
    HROO RST 5
    HROO U 40
    HROO U 1 A_FadeOut(0.05)
    wait
  Raise:
    stop
  }
}
